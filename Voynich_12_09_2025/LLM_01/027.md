        function computeNullFraction(text) {

            return nullChars / clean.length;

        function computeEntropy(freq, total) {
            let entropy = 0;
            for (let count of Object.values(freq)) {
                if (count > 0) {
                    const p = count / total;
                    entropy -= p * Math.log2(p);
                }
            }
            return entropy;
        }
        
        function computeHarmonics(text) {
            const clean = text.toLowerCase().replace(/[^a-z]/g, '');
            if (clean.length < 3) {
                return { alpha: 0, beta: 0, gamma: 0, null: 0 };
            }
            
            // Character frequency
            const charFreq = {};
            for (let c of clean) {
                charFreq[c] = (charFreq[c] || 0) + 1;
            }
            const total = clean.length;
            
            // Alpha: coherence (entropy-based)
            const expectedEntropy = 3.5;
            const actualEntropy = computeEntropy(charFreq, total);
            const alpha = Math.max(0, Math.min(1, 1 - Math.abs(actualEntropy - expectedEntropy) / expectedEntropy));
            
            // Beta: randomness (bigram entropy)
            const bigrams = {};
            for (let i = 0; i < clean.length - 1; i++) {
                const bigram = clean.substring(i, i + 2);
                bigrams[bigram] = (bigrams[bigram] || 0) + 1;
            }
            const bigramCount = Object.keys(bigrams).length;
            let beta = 0;
            if (bigramCount > 1) {
                const bigramEntropy = computeEntropy(bigrams, clean.length - 1);
                const maxBigramEntropy = Math.log2(bigramCount);
                beta = maxBigramEntropy > 0 ? bigramEntropy / maxBigramEntropy : 0;
            }
            
            // Gamma: repetition (uniqueness)
            const uniqueChars = Object.keys(charFreq).length;
            const gamma = 1 - (uniqueChars / total);
            
            // NULL: fraction of loop characters
            const nullFrac = computeNullFraction(clean);
            
            return { alpha, beta, gamma, null: nullFrac };
        }
